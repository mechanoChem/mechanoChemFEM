2  * @page growth Example 4 : Coupled mechano-chemical system with multiple doamins
3  * \section Introduction
4 * In this example, we first want to solve one diffusion equation:
5  * \f[
6  \frac{\partial C}{\partial t}+\nabla\cdot\boldsymbol{j}=0
7 \f]
8 * where \f$\boldsymbol{j}_1=-M\nabla C\f$ is the flux.
9 The boundary condition is
10 \f[
11 C=1 \text{ on }\Gamma_1; \quad \nabla \mu\cdot\boldsymbol{n}=jn \text{ on }\Gamma \backslash \Gamma_1
12 \f]
13  * Besides chemistry, we also solve elasticity problem at finite strain:
14  * \f[
15  \nabla\cdot\boldsymbol{T} = \boldsymbol{0}
16 \f]
17 *\f[
18 \boldsymbol{T}= \frac{1}{\det{\boldsymbol{F}^{\text{e}}}}\frac{\partial W}{\partial \boldsymbol{F}^{\text{e}}}\boldsymbol{F}^{\text{e}}
19 \f]
20  * To make it more interesting, we have mechanical deformation induced by species intecalation.
21  * \f[
22 \boldsymbol{F}=\boldsymbol{F}^{\text{e}}\boldsymbol{F}^{\text{g}}
23 \f]
24 *\f[
25 \boldsymbol{F}^{\text{g}}=\left(\frac{C}{C_\text{0}}\right)^{\frac{1}{3}}\mathbb{1}
26 \f]
27 In this example, we have two domains/materials. The diffusion equation is solved over the whole domains, while we restrict the mechanics into the one of the domain.
28 We want to model that the species transports from domain 1 into domain 2, and casues expansion of domain 2. By this example, we demonstrate how to setup
29 multiple domains, using <B>FE_Nothing</B> to exclude primary varialbe from certain doamin, and applying DOF constrains on the interface.

103 
104 we can calculate the flux as before, here we only discuss how to evaluate the deformation gradient tensor with isotropic growth.
105 First we need the total deformation graident tensor
106 *\code{.cpp}
107  deformationMap<Sacado::Fad::DFad<double>, dim> defMap(n_q_points);
108  getDeformationMap<Sacado::Fad::DFad<double>, dim>(fe_values, u_dof, ULocal, defMap);
109 \endcode
110 Then the deformation gradient tensor induced by elasticity can be computed via the inverse of multiplicative decomposition, and the stress can be
111 evaluated after chosing certain material model:
112 *\code{.cpp}
113  dealii::Table<3, Sacado::Fad::DFad<double> > P(n_q_points,dim,dim), Fe(n_q_points,dim,dim);
114 
115  for(unsigned int q=0; q<n_q_points;q++){
116  for (unsigned int i=0; i<dim; ++i){
117  for (unsigned int j=0; j<dim; ++j){
118  Fe[q][i][j]=defMap.F[q][i][j]/std::pow((c[q]/c_ini), 1.0/3.0); //Isotropic growth
119  }
120  }
121  }
122  this->ResidualEq.evaluateNeoHookeanStress(P, Fe);// NeoHookean model, Saint_Venant_Kirchhoff is also available

142 To apply Dirchlet boundary condition and contraints in general, we need to overload the <B>setup_constraints</B> function.
143 We can apply the Dirchlet boundary condition easily by <B>constraints</B>. Similarly
144 applying linear contraints on certain DOF can also be done by <B> constraints</B>, we only need to find the DOF where we want to apply the constraints:
145 *\code{.cpp}
146 template <int dim>
147 void growth<dim>::setup_constraints()
148 {
149  this->pcout<<"setup_constraints"<<std::endl;
150  hpFEM<dim>::constraints.clear ();
151  DoFTools::make_hanging_node_constraints (this->dof_handler, hpFEM<dim>::constraints);
152 
153  int totalDOF=this->totalDOF(this->primary_variables);
154  std::vector<bool> c_component (totalDOF, false); c_component[0]=true;
155  //apply constraints on boundary
156  VectorTools:: interpolate_boundary_values (hpFEM<dim>::dof_handler, dim, ZeroFunction<dim> (totalDOF),hpFEM<dim>::constraints, c_component);
157  //apply constraints on interface (domain 1 side)
158 
159  std::vector<types::global_dof_index> local_face_dof_indices_1 (this->fe_system[1]->dofs_per_face);
160  typename hp::DoFHandler<dim>::active_cell_iterator cell = this->dof_handler.begin_active(), endc=this->dof_handler.end();
161  for (;cell!=endc; ++cell){
162  if(cell->material_id()==1 ){
163  for (unsigned int f=0; f<GeometryInfo<dim>::faces_per_cell; ++f){
164  if (cell->at_boundary(f) == false){
165  if(cell->neighbor(f)->material_id()==0 and cell->neighbor(f)->has_children() == false){
166  cell->face(f)->get_dof_indices (local_face_dof_indices_1, 1);
167  for (unsigned int i=0; i<local_face_dof_indices_1.size(); ++i){
168  const unsigned int ck = this->fe_system[1]->face_system_to_component_index(i).first;
169  if(ck>0) hpFEM<dim>::constraints.add_line (local_face_dof_indices_1[i]);//add constrain line for all u dofs
170  }
171  }
172  }
173  }
174  }
175  }
176 
177  hpFEM<dim>::constraints.close ();
178 }
